cmake_minimum_required(VERSION 3.15)
project(Pong)

set(CMAKE_CXX_STANDARD 17)

# Source files
add_executable(Pong
    main.cpp
    Ball.cpp
    Paddle.cpp
)

if (WIN32)
    # -------------------------------------------------------
    # Windows — install raylib to C:\raylib with this layout:
    #   C:\raylib\include\raylib.h
    #   C:\raylib\lib\libraylib.a
    # -------------------------------------------------------
    target_include_directories(Pong PRIVATE "C:/raylib/include")
    target_link_directories(Pong PRIVATE "C:/raylib/lib")
    target_link_libraries(Pong raylib opengl32 gdi32 winmm)

elseif (APPLE)
    # -------------------------------------------------------
    # macOS — install raylib via Homebrew: brew install raylib
    # -------------------------------------------------------
    find_package(raylib REQUIRED)
    target_link_libraries(Pong raylib)

elseif (UNIX)
    # -------------------------------------------------------
    # Linux — install raylib via package manager or from source
    # -------------------------------------------------------
    find_package(raylib REQUIRED)
    target_link_libraries(Pong raylib m dl pthread)

endif()
